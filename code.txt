def correct_question(raw_question):
    """
    Uses LLM to:
    1. Fix spelling mistakes
    2. Fix grammar
    3. Form proper sentence
    4. Keep meaning same
    """
    correction_prompt = f"""You are a text correction assistant.
Fix the spelling mistakes, grammar errors and sentence formation 
in the following question. Keep the meaning exactly the same.
Only return the corrected question, nothing else.

Original question: {raw_question}

Corrected question:"""
    
    try:
        response = correction_llm.invoke(correction_prompt)
        corrected = response.content.strip()
        return corrected
    except:
        return raw_question  # Return original if correction fails

def chat(user_message, history):
    """
    Handles chat with:
    1. Auto correction of user message
    2. RAG retrieval with chat history
    3. Source document display
    """
    if not user_message.strip():
        return history, history, "", ""

    # Step 1: Auto correct the question
    corrected_message = correct_question(user_message)
    
    # Show correction if different
    correction_note = ""
    if corrected_message.lower() != user_message.lower():
        correction_note = f"✏️ Corrected to: \"{corrected_message}\""

    # Step 2: Get answer from RAG
    try:
        result = qa_chain.invoke({"question": corrected_message})
        answer = result["answer"]
        
        # Get source files
        sources = set()
        if "source_documents" in result:
            for doc in result["source_documents"]:
                src = doc.metadata.get("source", "")
                if src:
                    sources.add(os.path.basename(src))
        
        # Format final answer
        final_answer = answer
        if sources:
            final_answer += f"\n\n📄 Sources: {', '.join(list(sources)[:3])}"
        if correction_note:
            final_answer = correction_note + "\n\n" + final_answer

    except Exception as e:
        final_answer = f"Error: {str(e)}"
        corrected_message = user_message

    # Update history
    history.append({"role": "user", "content": user_message})
    history.append({"role": "assistant", "content": final_answer})
    
    return history, history, corrected_message, ""


def clear_chat():
    """Clear chat history and memory"""
    memory.clear()
    return [], [], "", ""

sessions = {}

def create_new_session():
    sid = str(uuid.uuid4())[:8]
    sessions[sid] = {
        "title": "New Chat",
        # clean_messages stores ONLY plain text, no correction notes, no source lines
        # used to build chat_history for the LLM
        "clean_messages": [],
        # display_messages stores the full formatted text shown to user
        "display_messages": [],
        "created_at": datetime.now().strftime("%b %d, %H:%M")
    }
    return sid

# ----------------------------------------
# STEP 6: Core Functions
# ----------------------------------------
def correct_question(raw_question):
    correction_prompt = f"""You are a text correction assistant.
Fix the spelling mistakes, grammar errors and sentence formation 
in the following question. Keep the meaning exactly the same.
Only return the corrected question, nothing else.

Original question: {raw_question}
Corrected question:"""
    try:
        response = correction_llm.invoke(correction_prompt)
        return response.content.strip()
    except:
        return raw_question


def build_history_string(clean_messages):
    """
    Build a readable history string from clean messages list.
    clean_messages = [{"role": "user"/"assistant", "content": "plain text"}, ...]
    Returns a formatted string like:
        Human: ...
        Assistant: ...
    """
    if not clean_messages:
        return "No previous conversation."
    lines = []
    for m in clean_messages:
        prefix = "Human" if m["role"] == "user" else "Assistant"
        lines.append(f"{prefix}: {m['content']}")
    return "\n".join(lines)


def build_history_tuples(clean_messages):
    """
    Build list of (human, ai) tuples for ConversationalRetrievalChain.
    """
    tuples = []
    for i in range(0, len(clean_messages) - 1, 2):
        if i + 1 < len(clean_messages):
            h = clean_messages[i]["content"] if clean_messages[i]["role"] == "user" else ""
            a = clean_messages[i+1]["content"] if clean_messages[i+1]["role"] == "assistant" else ""
            if h and a:
                tuples.append((h, a))
    return tuples


def chat(user_message, current_sid, all_sessions_json):
    if not user_message.strip():
        return gr.update(), current_sid, all_sessions_json, "", ""

    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)

    if not current_sid or current_sid not in sessions:
        current_sid = create_new_session()

    # Auto correct
    corrected_message = correct_question(user_message)
    correction_note = ""
    if corrected_message.lower() != user_message.lower():
        correction_note = f"✏️ Corrected: \"{corrected_message}\""

    # Build clean history for LangChain — ONLY plain Q&A, no formatting
    clean_msgs = sessions[current_sid]["clean_messages"]
    chat_history_tuples = build_history_tuples(clean_msgs)
    chat_history_string = build_history_string(clean_msgs)

    # RAG call — pass both tuple format (for chain) and string (for prompt)
    try:
        result = qa_chain.invoke({
            "question": corrected_message,
            "chat_history": chat_history_tuples,
            # This goes into the prompt template's {chat_history} variable
            # We override by including it in the question context below
        })
        answer = result["answer"]

        sources = set()
        if "source_documents" in result:
            for doc in result["source_documents"]:
                src = doc.metadata.get("source", "")
                if src:
                    sources.add(os.path.basename(src))

        # Build display answer (with sources and correction note)
        display_answer = answer
        if sources:
            display_answer += f"\n\n📄 *Sources: {', '.join(list(sources)[:3])}*"
        if correction_note:
            display_answer = correction_note + "\n\n" + display_answer

    except Exception as e:
        answer = f"Error: {str(e)}"
        display_answer = f"❌ Error: {str(e)}"

    # Save CLEAN messages (plain text only, for history)
    sessions[current_sid]["clean_messages"].append(
        {"role": "user", "content": corrected_message}
    )
    sessions[current_sid]["clean_messages"].append(
        {"role": "assistant", "content": answer}
    )

    # Save DISPLAY messages (formatted, for UI)
    sessions[current_sid]["display_messages"].append(
        {"role": "user", "content": user_message}
    )
    sessions[current_sid]["display_messages"].append(
        {"role": "assistant", "content": display_answer}
    )

    # Auto-title from first question
    if len(sessions[current_sid]["clean_messages"]) == 2:
        sessions[current_sid]["title"] = corrected_message[:38].strip()

    # Build Gradio display format
    display_msgs = [{"role": m["role"], "content": m["content"]}
                    for m in sessions[current_sid]["display_messages"]]

    sessions_json = json.dumps(sessions)
    return display_msgs, current_sid, sessions_json, correction_note, ""


def build_choices():
    return [(f"💬 {data['title']}  •  {data['created_at']}", sid)
            for sid, data in reversed(list(sessions.items()))]


def new_chat(all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    sid = create_new_session()
    sessions_json = json.dumps(sessions)
    return [], sid, sessions_json, "", gr.update(choices=build_choices(), value=sid)


def load_chat(sid, all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if not sid or sid not in sessions:
        return [], sid, all_sessions_json, ""
    display_msgs = [{"role": m["role"], "content": m["content"]}
                    for m in sessions[sid]["display_messages"]]
    return display_msgs, sid, all_sessions_json, ""


def delete_chat(current_sid, all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if current_sid in sessions:
        del sessions[current_sid]
    if sessions:
        new_sid = list(sessions.keys())[-1]
    else:
        new_sid = create_new_session()
    display_msgs = [{"role": m["role"], "content": m["content"]}
                    for m in sessions[new_sid]["display_messages"]]
    sessions_json = json.dumps(sessions)
    return display_msgs, new_sid, sessions_json, "", gr.update(choices=build_choices(), value=new_sid)


def handle_chat(user_message, current_sid, sessions_json):
    msgs, sid, sj, corr, _ = chat(user_message, current_sid, sessions_json)
    return msgs, sid, sj, corr, "", gr.update(choices=build_choices(), value=sid)


def handle_new_chat(sessions_json):
    msgs, sid, sj, _, dropdown = new_chat(sessions_json)
    return msgs, sid, sj, "", dropdown


def handle_load_chat(sid, sessions_json):
    if not sid:
        return gr.update(), gr.update(), sessions_json, ""
    msgs, new_sid, sj, _ = load_chat(sid, sessions_json)
    return msgs, new_sid, sj, ""


def handle_delete_chat(current_sid, sessions_json):
    msgs, sid, sj, _, dropdown = delete_chat(current_sid, sessions_json)
    return msgs, sid, sj, "", dropdown

def correct_question(raw_question):
    correction_prompt = f"""Fix spelling and grammar only. Keep meaning same.
Return only the corrected question, nothing else.
Original: {raw_question}
Corrected:"""
    try:
        response = correction_llm.invoke(correction_prompt)
        return response.content.strip()
    except:
        return raw_question


def is_table_request(question):
    keywords = [
        "table", "tabular", "in table", "as table",
        "compare", "comparison", "vs", "versus",
        "side by side", "list all", "show all rates",
        "differences between", "tabulate"
    ]
    return any(kw in question.lower() for kw in keywords)


def build_history_tuples(clean_messages):
    tuples = []
    for i in range(0, len(clean_messages) - 1, 2):
        if i + 1 < len(clean_messages):
            h = clean_messages[i]["content"] if clean_messages[i]["role"] == "user" else ""
            a = clean_messages[i+1]["content"] if clean_messages[i+1]["role"] == "assistant" else ""
            if h and a:
                tuples.append((h, a))
    return tuples


def get_answer(question, chat_history_tuples, use_table=False):
    no_data_phrases = [
        "don't have enough data", "not explicitly mentioned",
        "not mentioned in", "cannot find", "no information",
        "not available", "do not have", "i don't have",
        "no specific", "not found"
    ]

    chain = qa_chain_table if use_table else qa_chain

    # Try 1: Direct question
    try:
        result = chain.invoke({"question": question, "chat_history": chat_history_tuples})
        answer = result["answer"]
        sources = result.get("source_documents", [])
        is_weak = any(p in answer.lower() for p in no_data_phrases)
        if not is_weak:
            return answer, sources
    except Exception as e:
        print(f"❌ Try 1 failed: {type(e).__name__}: {e}")

    # Try 2: Inject nearby areas
    found_areas, nearby = get_nearby_areas(question.lower())
    if nearby:
        nearby_str = ", ".join(nearby[:6])
        enriched_q = (
            f"{question} "
            f"Also search for data from these nearby areas: {nearby_str}. "
            f"Use any available data from these areas to provide a useful answer."
        )
        try:
            result2 = chain.invoke({"question": enriched_q, "chat_history": chat_history_tuples})
            answer2 = result2["answer"]
            sources2 = result2.get("source_documents", [])
            is_weak2 = any(p in answer2.lower() for p in no_data_phrases)
            if not is_weak2:
                area_note = ", ".join([a.title() for a in found_areas]) if found_areas else "requested area"
                prefix = f"📍 *Nearby area data included for **{area_note}**:*\n\n"
                return prefix + answer2, sources2
        except Exception as e:
            print(f"❌ Try 2 failed: {type(e).__name__}: {e}")

    # Try 3: General Hyderabad fallback
    general_q = (
        f"What are the property rates, rental trends and market data "
        f"available in Hyderabad? The user asked about: '{question}'. "
        f"Provide whatever relevant data is available."
    )
    try:
        result3 = chain.invoke({"question": general_q, "chat_history": chat_history_tuples})
        answer3 = result3["answer"]
        sources3 = result3.get("source_documents", [])
        return f"📍 *Showing general Hyderabad data:*\n\n{answer3}", sources3
    except Exception as e:
        print(f"❌ Try 3 failed: {type(e).__name__}: {e}")

    return (
        "Unable to retrieve data. Please rephrase or ask about a specific area "
        "like Kondapur, Kukatpally, Jadcherla, or Gandipet.",
        []
    )

def chat(user_message, current_sid, all_sessions_json):
    if not user_message.strip():
        return gr.update(), current_sid, all_sessions_json, "", ""

    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)

    if not current_sid or current_sid not in sessions:
        current_sid = create_new_session()

    corrected_message = correct_question(user_message)
    correction_note = ""
    if corrected_message.lower() != user_message.lower():
        correction_note = f"✏️ Corrected: \"{corrected_message}\""

    clean_msgs = sessions[current_sid]["clean_messages"]
    chat_history_tuples = build_history_tuples(clean_msgs)
    use_table = is_table_request(corrected_message)

    try:
        answer, _ = get_answer(corrected_message, chat_history_tuples, use_table=use_table)
        display_answer = answer
        if correction_note:
            display_answer = f"*{correction_note}*\n\n" + display_answer
    except Exception as e:
        answer = f"Error: {str(e)}"
        display_answer = f"❌ **Error:** {str(e)}"

    sessions[current_sid]["clean_messages"].append({"role": "user", "content": corrected_message})
    sessions[current_sid]["clean_messages"].append({"role": "assistant", "content": answer})
    sessions[current_sid]["display_messages"].append({"role": "user", "content": user_message})
    sessions[current_sid]["display_messages"].append({"role": "assistant", "content": display_answer})

    if len(sessions[current_sid]["clean_messages"]) == 2:
        sessions[current_sid]["title"] = corrected_message[:38].strip()

    display_msgs = [{"role": m["role"], "content": m["content"]}
                    for m in sessions[current_sid]["display_messages"]]

    return display_msgs, current_sid, json.dumps(sessions), correction_note, ""

def build_choices():
    return [(f"{data['title']}  ·  {data['created_at']}", sid)
            for sid, data in reversed(list(sessions.items()))]

def new_chat(all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    sid = create_new_session()
    return [], sid, json.dumps(sessions), "", gr.update(choices=build_choices(), value=sid)

def load_chat(sid, all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if not sid or sid not in sessions:
        return [], sid, all_sessions_json, ""
    display_msgs = [{"role": m["role"], "content": m["content"]}
                    for m in sessions[sid]["display_messages"]]
    return display_msgs, sid, all_sessions_json, ""

def delete_chat(current_sid, all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if current_sid in sessions:
        del sessions[current_sid]
    new_sid = list(sessions.keys())[-1] if sessions else create_new_session()
    display_msgs = [{"role": m["role"], "content": m["content"]}
                    for m in sessions[new_sid]["display_messages"]]
    return display_msgs, new_sid, json.dumps(sessions), "", gr.update(choices=build_choices(), value=new_sid)

def handle_chat(user_message, current_sid, sessions_json):
    msgs, sid, sj, corr, _ = chat(user_message, current_sid, sessions_json)
    return msgs, sid, sj, corr, "", gr.update(choices=build_choices(), value=sid)

def handle_new_chat(sessions_json):
    msgs, sid, sj, _, dd = new_chat(sessions_json)
    return msgs, sid, sj, "", dd

def handle_load_chat(sid, sessions_json):
    if not sid:
        return gr.update(), gr.update(), sessions_json, ""
    msgs, new_sid, sj, _ = load_chat(sid, sessions_json)
    return msgs, new_sid, sj, ""

def handle_delete_chat(current_sid, sessions_json):
    msgs, sid, sj, _, dd = delete_chat(current_sid, sessions_json)
    return msgs, sid, sj, "", dd

def correct_question(raw_question):
    try:
        response = correction_llm.invoke(
            f"Fix spelling and grammar only. Keep meaning same.\n"
            f"Return only the corrected question, nothing else.\n"
            f"Original: {raw_question}\nCorrected:"
        )
        return response.content.strip()
    except:
        return raw_question

def is_table_request(question):
    keywords = ["table","tabular","in table","as table","compare","comparison",
                "vs","versus","side by side","list all","show all rates",
                "differences between","tabulate"]
    return any(kw in question.lower() for kw in keywords)

def build_history_tuples(clean_messages):
    tuples = []
    for i in range(0, len(clean_messages) - 1, 2):
        if i + 1 < len(clean_messages):
            h = clean_messages[i]["content"]   if clean_messages[i]["role"]   == "user"      else ""
            a = clean_messages[i+1]["content"] if clean_messages[i+1]["role"] == "assistant" else ""
            if h and a:
                tuples.append((h, a))
    return tuples

def get_answer(question, chat_history_tuples, use_table=False):
    no_data_phrases = [
        "don't have enough data","not explicitly mentioned","not mentioned in",
        "cannot find","no information","not available","do not have",
        "i don't have","no specific","not found"
    ]
    chain = qa_chain_table if use_table else qa_chain

    try:
        result = chain.invoke({"question": question, "chat_history": chat_history_tuples})
        answer, sources = result["answer"], result.get("source_documents", [])
        if not any(p in answer.lower() for p in no_data_phrases):
            return answer, sources
    except Exception as e:
        print(f"❌ Try 1 failed: {type(e).__name__}: {e}")

    found_areas, nearby = get_nearby_areas(question.lower())
    if nearby:
        enriched_q = (f"{question} Also search for data from these nearby areas: "
                      f"{', '.join(nearby[:6])}. Use any available data to provide a useful answer.")
        try:
            result2 = chain.invoke({"question": enriched_q, "chat_history": chat_history_tuples})
            answer2, sources2 = result2["answer"], result2.get("source_documents", [])
            if not any(p in answer2.lower() for p in no_data_phrases):
                area_note = ", ".join([a.title() for a in found_areas]) if found_areas else "requested area"
                return f"📍 *Nearby area data for **{area_note}**:*\n\n{answer2}", sources2
        except Exception as e:
            print(f"❌ Try 2 failed: {type(e).__name__}: {e}")

    try:
        general_q = (f"What are the property rates, rental trends and market data "
                     f"available in Hyderabad? The user asked about: '{question}'. "
                     f"Provide whatever relevant data is available.")
        result3 = chain.invoke({"question": general_q, "chat_history": chat_history_tuples})
        return f"📍 *Showing general Hyderabad data:*\n\n{result3['answer']}", result3.get("source_documents", [])
    except Exception as e:
        print(f"❌ Try 3 failed: {type(e).__name__}: {e}")

    return ("Unable to retrieve data. Please rephrase or ask about a specific area.", [])

def chat(user_message, current_sid, all_sessions_json):
    if not user_message.strip():
        return gr.update(), current_sid, all_sessions_json, "", ""

    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if not current_sid or current_sid not in sessions:
        current_sid = create_new_session()

    corrected_message = correct_question(user_message)
    correction_note = f"✏️ Corrected: \"{corrected_message}\"" if corrected_message.lower() != user_message.lower() else ""

    chat_history_tuples = build_history_tuples(sessions[current_sid]["clean_messages"])
    use_table = is_table_request(corrected_message)

    try:
        answer, _ = get_answer(corrected_message, chat_history_tuples, use_table=use_table)
        display_answer = (f"*{correction_note}*\n\n" + answer) if correction_note else answer
    except Exception as e:
        answer = f"Error: {str(e)}"
        display_answer = f"❌ **Error:** {str(e)}"

    sessions[current_sid]["clean_messages"]  += [{"role":"user","content":corrected_message}, {"role":"assistant","content":answer}]
    sessions[current_sid]["display_messages"] += [{"role":"user","content":user_message}, {"role":"assistant","content":display_answer}]

    if len(sessions[current_sid]["clean_messages"]) == 2:
        sessions[current_sid]["title"] = corrected_message[:38].strip()

    display_msgs = [{"role": m["role"], "content": m["content"]} for m in sessions[current_sid]["display_messages"]]
    return display_msgs, current_sid, json.dumps(sessions), correction_note, ""

def build_choices():
    return [(f"{d['title']}  ·  {d['created_at']}", sid) for sid, d in reversed(list(sessions.items()))]

def new_chat(all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    sid = create_new_session()
    return [], sid, json.dumps(sessions), "", gr.update(choices=build_choices(), value=sid)

def load_chat(sid, all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if not sid or sid not in sessions:
        return [], sid, all_sessions_json, ""
    msgs = [{"role": m["role"], "content": m["content"]} for m in sessions[sid]["display_messages"]]
    return msgs, sid, all_sessions_json, ""

def delete_chat(current_sid, all_sessions_json):
    global sessions
    if all_sessions_json and all_sessions_json != "{}":
        sessions = json.loads(all_sessions_json)
    if current_sid in sessions:
        del sessions[current_sid]
    new_sid = list(sessions.keys())[-1] if sessions else create_new_session()
    msgs = [{"role": m["role"], "content": m["content"]} for m in sessions[new_sid]["display_messages"]]
    return msgs, new_sid, json.dumps(sessions), "", gr.update(choices=build_choices(), value=new_sid)

def handle_chat(user_message, current_sid, sessions_json):
    msgs, sid, sj, corr, _ = chat(user_message, current_sid, sessions_json)
    return msgs, sid, sj, corr, "", gr.update(choices=build_choices(), value=sid)

def handle_new_chat(sessions_json):
    msgs, sid, sj, _, dd = new_chat(sessions_json)
    return msgs, sid, sj, "", dd

def handle_load_chat(sid, sessions_json):
    if not sid:
        return gr.update(), gr.update(), sessions_json, ""
    msgs, new_sid, sj, _ = load_chat(sid, sessions_json)
    return msgs, new_sid, sj, ""

def handle_delete_chat(current_sid, sessions_json):
    msgs, sid, sj, _, dd = delete_chat(current_sid, sessions_json)
    return msgs, sid, sj, "", dd
